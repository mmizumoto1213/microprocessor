--- BEGINNING OF SHIFT-AND-ADD MULTIPLIER ---
# RESET R0 TO BE ABLE FOR INCREMENTATION
IMM 00000001
# ADD THE CURRENT INDEX OF THE MULTIPLICAND
ADD R7
# MOVE IT TO R15 TO ACCESS
MOV R15
# MOVE IT RO R7 TO UPDATE INDEX
MOV R7
# GET THE VALUE OF THE MULTIPLICAND
LW R0
# MOVE IT TO CORRECT REGISTER
MOV R1
# RESET R0 TO BE ABLE FOR INCREMENTATION
IMM 00000001
# ADD THE CURRENT INDEX OF THE MULTIPLICAND
ADD R7
# MOVE IT TO R15 TO ACCESS
MOV R15
# MOVE IT RO R7 TO UPDATE INDEX
MOV R7
# GET THE VALUE OF THE MULTIPLICAND
LW R0
# MOVE IT TO CORRECT REGISTER
MOV R2
# LOAD IN FIRST HALF OF MULTIPLIER
IMM 00000001
# ADD THE CURRENT INDEX
ADD R7
# MOVE IT TO CORRECT REGISTER
MOV R15
# UPDATE THE CURRENT INDEX
MOV R7
# LOAD IN THE VALUE
LW R0
# MOVE IT TO STORAGE SPACE
MOV R3
# LOAD IN SECOND HALF OF MULTIPLER
IMM 00000001
# ADD THE CURRENT INDEX
ADD R7
# UPDATE THE CURRENT INDEX
MOV R7
# MOVE IT TO CORRECT REGISTER
MOV R15
# LOAD IN THE VALUE
LW R0
# MOVE IT TO STORAGE SPACE
MOV R4
# SET R0 TO 1 IN ORDER TO CHECK FOR SHIFT-AND-ADD
IMM 00000001
# IF IT IS 1 BRANCH TO THE ADDING FUNCTION
BEQ R3
# RESET R0 TO 1 FOR INCREMENTING ANSWER INDEX
IMM 00000001
# ADD THE INDEX OF THE ANSWER
ADD R9
# MOVE IT BACK TO THE CORRECT REGISTER FOR USE LATER
MOV R9
# RESET R0 TO GET THE NEXT BIT OF MULTIPLIER READY FOR INCREMENTING
IMM 00000001
# ADD THE CURRENT INDEX OF THE BIT
ADD R4
# MOVE IT BACK TO R4
MOV R4
# GET THE IMMEDIATE OF THE START OF THE LOOP
IMM 00000000
# MOVE IT TO R14 FOR ACCESS
MOV R14
# RESET R0 TO THE END OF THE LOOP
IMM 01000000
# IF GREATER THAN THE CURRENT INDEX WE JUMP BACK
BGE R4
# OTHERWISE SET R0 TO THE END OF THE FUNCTION IN THE LOOKUP TABLE INDEX
IMM 00000010
# MOVE IT TO R14 FOR THE LOOKUP TABLE
MOV R14
# JUMP TO THE END AS THE LOOP IS NOW FINISHED
BEQ R0

--- BEGINNING OF ADDING FUNCTION ---
# RESET R0 TO BE ABLE TO MOVE IN THE CURRENT MULTIPLIER
IMM 00000000
# MOVE IN THE FIRST HALF OF THE MULTIPLICAND
ADD R1
# SHIFT IT LEFT BY 1
SBL R0
# MOVE IT BACK TO REGISTER 1
MOV R1
# RESET R0 TO BE ABLE TO MOVE IN THE SECOND PART OF THE MULTIPLIER
IMM 00000000
# MOVE IN THE SECOND PART OF THE MULTIPLIER
ADD R2
# SHIFT IT LEFT BY 1
SBL R0
# MOVE IT BACK TO REGISTER 2
SBL R2
# LOAD THE JUMP BACK TO THE MIDDLE OF THE SHIFT-AND-ADD LOOP
IMM 00000001
# MOVE IT TO THE LOOK UP TABLE INDEX
MOVE R14
# JUMP BACK
BEQ R0
